import { useRouter } from 'expo-router';
import React, { useState } from 'react';
import { Alert, KeyboardAvoidingView, Platform, StyleSheet, Text, View } from 'react-native';
import AuthEntry from '@/components/login/AuthEntry';
import VerificationForm from '@/components/login/VerificationForm';
import { useAuth } from '../../src/domain/auth';

type AuthMethod = 'email' | 'phone' | null;

export default function LoginScreen() {
    const [authMethod, setAuthMethod] = useState<AuthMethod>(null);
    const [email, setEmail] = useState('');
    const [phoneNumber, setPhoneNumber] = useState('');
    const [verificationCode, setVerificationCode] = useState('');
    const [isVerificationMode, setIsVerificationMode] = useState(true);
    const { loginWithEmail, loginWithPhone, loginWithOAuth, verifyAccount, isLoading, error, clearError } = useAuth();
    const router = useRouter();

    const handleEmailLogin = async () => {
        if (!email.trim()) {
            Alert.alert('Error', 'Please enter your email address');
            return;
        }

        try {
            clearError();
            await loginWithEmail(email);
            // If we get here, login was successful
            router.replace('/(tabs)/home' as any);
        } catch (error) {
            if (
                error instanceof Error &&
                error.message === 'Account verification required'
            ) {
                setIsVerificationMode(true);
            } else {
                Alert.alert(
                    'Login Failed',
                    error instanceof Error ? error.message : 'An error occurred'
                );
            }
        }
    };

    const handlePhoneLogin = async () => {
        if (!phoneNumber.trim()) {
            Alert.alert('Error', 'Please enter your phone number');
            return;
        }

        try {
            clearError();
            await loginWithPhone(phoneNumber);
            // If we get here, login was successful
            router.replace('/(tabs)/home' as any);
        } catch (error) {
            if (
                error instanceof Error &&
                error.message === 'Account verification required'
            ) {
                setIsVerificationMode(true);
            } else {
                Alert.alert(
                    'Login Failed',
                    error instanceof Error ? error.message : 'An error occurred'
                );
            }
        }
    };

    const handleGoogleLogin = async () => {
        try {
            clearError();
            await loginWithOAuth('google');
            // If we get here, login was successful
            router.replace('/(tabs)/home' as any);
        } catch (error) {
            Alert.alert(
                'Login Failed',
                error instanceof Error ? error.message : 'An error occurred'
            );
        }
    };

    const handleVerification = async () => {
        if (!verificationCode.trim()) {
            Alert.alert('Error', 'Please enter the verification code');
            return;
        }

        try {
            clearError();
            await verifyAccount(verificationCode);
            // If we get here, verification was successful
            router.replace('/(tabs)/home' as any);
        } catch (error) {
            Alert.alert(
                'Verification Failed',
                error instanceof Error ? error.message : 'An error occurred'
            );
        }
    };

    const handleBackToLogin = () => {
        setIsVerificationMode(false);
        setVerificationCode('');
        setAuthMethod(null);
        clearError();
    };

    const handleSelectAuthMethod = (method: 'email' | 'phone') => {
        setAuthMethod(method);
        setIsVerificationMode(false);
        setEmail('');
        setPhoneNumber('');
        setVerificationCode('');
        clearError();
    };

    return (
        <KeyboardAvoidingView
            style={styles.container}
            behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
        >
            <View style={styles.content}>
                <Text style={styles.title}>Welcome to Rampa</Text>
                <Text style={styles.subtitle}>
                    {isVerificationMode
                        ? `Enter the verification code sent to your ${authMethod === 'email' ? 'email' : 'phone'}`
                        : 'Your secure remittance platform'}
                </Text>

                {!isVerificationMode ? (
                    <AuthEntry
                        authMethod={authMethod}
                        onSelectMethod={handleSelectAuthMethod}
                        email={email}
                        onChangeEmail={setEmail}
                        phoneNumber={phoneNumber}
                        onChangePhone={setPhoneNumber}
                        onEmailLogin={handleEmailLogin}
                        onPhoneLogin={handlePhoneLogin}
                        onGoogleLogin={handleGoogleLogin}
                        onBack={() => setAuthMethod(null)}
                        isLoading={isLoading}
                        error={error ?? null}
                    />
                ) : (
                    <VerificationForm
                        code={verificationCode}
                        onChangeCode={setVerificationCode}
                        onVerify={handleVerification}
                        onBackToLogin={handleBackToLogin}
                        isLoading={isLoading}
                        error={error ?? null}
                    />
                )}

                <View style={styles.footer}>
                    <Text style={styles.footerText}>
                        By continuing, you agree to our Terms of Service and
                        Privacy Policy
                    </Text>
                </View>
            </View>
        </KeyboardAvoidingView>
    );
}

const styles = StyleSheet.create({
    container: {
        flex: 1,
        backgroundColor: '#ffffff',
    },
    content: {
        flex: 1,
        paddingHorizontal: 24,
        paddingTop: 60,
        paddingBottom: 40,
        justifyContent: 'center',
    },
    title: {
        fontSize: 32,
        fontWeight: 'bold',
        color: '#1a1a1a',
        textAlign: 'center',
        marginBottom: 8,
    },
    subtitle: {
        fontSize: 16,
        color: '#666666',
        textAlign: 'center',
        marginBottom: 40,
        lineHeight: 24,
    },
    // form-related styles moved into components/login
    footer: {
        marginTop: 40,
        paddingHorizontal: 20,
    },
    footerText: {
        fontSize: 12,
        color: '#999999',
        textAlign: 'center',
        lineHeight: 18,
    },
});
